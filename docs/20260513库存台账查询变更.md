## 2. 示例 SQL

库存台账查询示例SQL：

```SQL
DECLARE @cInvCode NVARCHAR(50) = '3314010002';
DECLARE @StartDate DATETIME = '2020-04-01';
DECLARE @EndDate DATETIME = '2026-05-12';

WITH AllFlow AS (
    -- 期初结存
    SELECT
        R.cWhCode AS 仓库编码,
        CASE
            WHEN R.dnmaketime IS NOT NULL THEN R.dnmaketime
            ELSE CAST(R.dDate AS DATETIME)
        END AS 单据日期,
        CAST(R.dVeriDate AS DATETIME) AS 审核日期,
        CASE
            WHEN R.dnverifytime IS NOT NULL THEN R.dnverifytime
            ELSE CAST(R.dVeriDate AS DATETIME)
        END AS 审核排序时间,
        R.cCode AS 单据号,
        N'期初结存' AS 单据类型,
        CAST(ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)) AS 收入数量,
        CAST(0 AS DECIMAL(18,6)) AS 发出数量,
        CAST(ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)) AS 数量变化,
        CAST(RS.AutoID AS BIGINT) AS 明细ID
    FROM rdrecord34 R
    INNER JOIN rdrecords34 RS ON R.ID = RS.ID
    WHERE RS.cInvCode = @cInvCode
      AND R.dVeriDate < DATEADD(DAY, 1, @EndDate)

    UNION ALL

    -- 采购入库
    SELECT
        R.cWhCode,
        CASE WHEN R.dnmaketime IS NOT NULL THEN R.dnmaketime ELSE CAST(R.dDate AS DATETIME) END,
        CAST(R.dVeriDate AS DATETIME),
        CASE WHEN R.dnverifytime IS NOT NULL THEN R.dnverifytime ELSE CAST(R.dVeriDate AS DATETIME) END,
        R.cCode,
        ISNULL(S.cRdName, N'采购入库'),
        CAST(ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(0 AS DECIMAL(18,6)),
        CAST(ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(RS.AutoID AS BIGINT)
    FROM RdRecord01 R
    INNER JOIN rdrecords01 RS ON R.ID = RS.ID
    LEFT JOIN Rd_Style S ON R.cRdCode = S.cRdCode
    WHERE RS.cInvCode = @cInvCode
      AND R.dVeriDate < DATEADD(DAY, 1, @EndDate)

    UNION ALL

    -- 产成品入库
    SELECT
        R.cWhCode,
        CASE WHEN R.dnmaketime IS NOT NULL THEN R.dnmaketime ELSE CAST(R.dDate AS DATETIME) END,
        CAST(R.dVeriDate AS DATETIME),
        CASE WHEN R.dnverifytime IS NOT NULL THEN R.dnverifytime ELSE CAST(R.dVeriDate AS DATETIME) END,
        R.cCode,
        ISNULL(S.cRdName, N'产成品入库'),
        CAST(ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(0 AS DECIMAL(18,6)),
        CAST(ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(RS.AutoID AS BIGINT)
    FROM rdrecord10 R
    INNER JOIN rdrecords10 RS ON R.ID = RS.ID
    LEFT JOIN Rd_Style S ON R.cRdCode = S.cRdCode
    WHERE RS.cInvCode = @cInvCode
      AND R.dVeriDate < DATEADD(DAY, 1, @EndDate)

    UNION ALL

    -- 其他入库 / 调拨入库 / 盘盈入库 / 形态转换入 / 二部倒冲入库
    SELECT
        R.cWhCode,
        CASE WHEN R.dnmaketime IS NOT NULL THEN R.dnmaketime ELSE CAST(R.dDate AS DATETIME) END,
        CAST(R.dVeriDate AS DATETIME),
        CASE WHEN R.dnverifytime IS NOT NULL THEN R.dnverifytime ELSE CAST(R.dVeriDate AS DATETIME) END,
        R.cCode,
        ISNULL(S.cRdName, N'其他入库'),
        CAST(ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(0 AS DECIMAL(18,6)),
        CAST(ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(RS.AutoID AS BIGINT)
    FROM RdRecord08 R
    INNER JOIN rdrecords08 RS ON R.ID = RS.ID
    LEFT JOIN Rd_Style S ON R.cRdCode = S.cRdCode
    WHERE RS.cInvCode = @cInvCode
      AND R.dVeriDate < DATEADD(DAY, 1, @EndDate)

    UNION ALL

    -- 其他出库 / 调拨出库 / 盘亏出库 / 外协加工出库 / 形态转换出
    SELECT
        R.cWhCode,
        CASE WHEN R.dnmaketime IS NOT NULL THEN R.dnmaketime ELSE CAST(R.dDate AS DATETIME) END,
        CAST(R.dVeriDate AS DATETIME),
        CASE WHEN R.dnverifytime IS NOT NULL THEN R.dnverifytime ELSE CAST(R.dVeriDate AS DATETIME) END,
        R.cCode,
        ISNULL(S.cRdName, N'其他出库'),
        CAST(0 AS DECIMAL(18,6)),
        CAST(ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(-ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(RS.AutoID AS BIGINT)
    FROM RdRecord09 R
    INNER JOIN rdrecords09 RS ON R.ID = RS.ID
    LEFT JOIN Rd_Style S ON R.cRdCode = S.cRdCode
    WHERE RS.cInvCode = @cInvCode
      AND R.dVeriDate < DATEADD(DAY, 1, @EndDate)

    UNION ALL

    -- 材料/领用类出库
    SELECT
        R.cWhCode,
        CASE WHEN R.dnmaketime IS NOT NULL THEN R.dnmaketime ELSE CAST(R.dDate AS DATETIME) END,
        CAST(R.dVeriDate AS DATETIME),
        CASE WHEN R.dnverifytime IS NOT NULL THEN R.dnverifytime ELSE CAST(R.dVeriDate AS DATETIME) END,
        R.cCode,
        ISNULL(S.cRdName, N'材料出库'),
        CAST(0 AS DECIMAL(18,6)),
        CAST(ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(-ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(RS.AutoID AS BIGINT)
    FROM rdrecord11 R
    INNER JOIN rdrecords11 RS ON R.ID = RS.ID
    LEFT JOIN Rd_Style S ON R.cRdCode = S.cRdCode
    WHERE RS.cInvCode = @cInvCode
      AND R.dVeriDate < DATEADD(DAY, 1, @EndDate)

    UNION ALL

    -- 销售出库
    SELECT
        R.cWhCode,
        CASE WHEN R.dnmaketime IS NOT NULL THEN R.dnmaketime ELSE CAST(R.dDate AS DATETIME) END,
        CAST(R.dVeriDate AS DATETIME),
        CASE WHEN R.dnverifytime IS NOT NULL THEN R.dnverifytime ELSE CAST(R.dVeriDate AS DATETIME) END,
        R.cCode,
        ISNULL(S.cRdName, N'销售出库'),
        CAST(0 AS DECIMAL(18,6)),
        CAST(ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(-ISNULL(RS.iQuantity, 0) AS DECIMAL(18,6)),
        CAST(RS.AutoID AS BIGINT)
    FROM rdrecord32 R
    INNER JOIN rdrecords32 RS ON R.ID = RS.ID
    LEFT JOIN Rd_Style S ON R.cRdCode = S.cRdCode
    WHERE RS.cInvCode = @cInvCode
      AND R.dVeriDate < DATEADD(DAY, 1, @EndDate)
),
Opening AS (
    SELECT ISNULL(SUM(数量变化), 0) AS 期初余额
    FROM AllFlow
    WHERE 审核日期 < @StartDate
),
PeriodFlow AS (
    SELECT *
    FROM AllFlow
    WHERE 审核日期 >= @StartDate
      AND 审核日期 < DATEADD(DAY, 1, @EndDate)
),
FlowWithSeq AS (
    SELECT
        ROW_NUMBER() OVER (
            ORDER BY 单据日期, 审核日期, 审核排序时间, 明细ID
        ) AS 序号,
        仓库编码,
        单据日期,
        审核日期,
        审核排序时间,
        单据号,
        单据类型,
        收入数量,
        发出数量,
        数量变化,
        明细ID
    FROM PeriodFlow
)
SELECT
    A.仓库编码,
    A.单据日期,
    A.审核日期,
    A.单据号,
    A.单据类型,
    A.收入数量,
    A.发出数量,
    O.期初余额 + ISNULL((
        SELECT SUM(B.数量变化)
        FROM FlowWithSeq B
        WHERE B.序号 <= A.序号
    ), 0) AS 结存数量
FROM FlowWithSeq A
CROSS JOIN Opening O
ORDER BY A.序号 DESC;

```
